## 基础用法

```tsx
import { Select } from "@/src/components/select";

const options = [
  { label: "北京", value: "beijing" },
  { label: "上海", value: "shanghai" },
];

export function Demo() {
  return (
    <Select
      options={options}
      placeholder="请选择城市"
      onChange={({ value }) => console.log(value)}
      onChangeValue={(value) => console.log(value)}
    />
  );
}
```
