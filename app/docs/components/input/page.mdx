## 基础用法

```tsx
import { Input } from "@/src/components/input";

export function Demo() {
  return (
    <Input
      placeholder="请输入内容"
      onChange={({ value, event }) => console.log(value, event)}
      onChangeValue={(value) => console.log(value)}
      onPressEnter={(event) => console.log(event.currentTarget.value)}
    />
  );
}
```

## 密码输入

```tsx
import { Input, Password } from "@/src/components/input";

export function Demo() {
  const { Password: InputPassword } = Input;

  return (
    <div className="grid gap-4">
      <Input.Password placeholder="请输入密码" />
      <Input.Password visibilityToggle={false} placeholder="不显示切换按钮" />
      <InputPassword placeholder="解构使用 Password" />
      <Password placeholder="命名导出使用 Password" />
    </div>
  );
}
```

## 多行文本

```tsx
import { Input, TextArea } from "@/src/components/input";

export function Demo() {
  const { TextArea: InputTextArea } = Input;

  return (
    <div className="grid gap-4">
      <Input.TextArea placeholder="请输入多行内容" rows={3} />
      <Input.TextArea autoSize={{ minRows: 2, maxRows: 6 }} />
      <InputTextArea placeholder="解构使用 TextArea" />
      <TextArea placeholder="命名导出使用 TextArea" />
    </div>
  );
}
```
